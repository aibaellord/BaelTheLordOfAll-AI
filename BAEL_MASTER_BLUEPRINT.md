# Bayesian reasoning for uncertainty

def bayesian_update(prior, likelihood, evidence):
    """
    P(H|E) = P(E|H) * P(H) / P(E)
    """
    posterior = (likelihood * prior) / evidence
    return posterior

def expected_utility(actions, outcomes, probabilities, utilities):
    """
    EU(a) = Σ P(o|a) * U(o)
    """
    expected_utilities = {}
    for action in actions:
        eu = 0
        for outcome, prob in zip(outcomes, probabilities[action]):
            eu += prob * utilities[outcome]
        expected_utilities[action] = eu
    return max(expected_utilities, key=expected_utilities.get)

# Nash equilibrium finding for multi-agent scenarios

def find_nash_equilibrium(players, strategies, payoffs):
    """
    Find strategy profiles where no player benefits from unilateral deviation
    """
    equilibria = []

    for profile in itertools.product(*strategies):
        is_equilibrium = True
        for i, player in enumerate(players):
            current_payoff = payoffs[profile]
            for alternative in strategies[i]:
                alt_profile = profile[:i] + (alternative,) + profile[i+1:]
                if payoffs[alt_profile] > current_payoff:
                    is_equilibrium = False
                    break
            if not is_equilibrium:
                break
        if is_equilibrium:
            equilibria.append(profile)

    return equilibria

def minimax(game_tree, depth, maximizing):
    """
    Minimax algorithm for adversarial scenarios
    """
    if depth == 0 or is_terminal(game_tree):
        return evaluate(game_tree)

    if maximizing:
        value = -infinity
        for child in game_tree.children:
            value = max(value, minimax(child, depth-1, False))
        return value
    else:
        value = infinity
        for child in game_tree.children:
            value = min(value, minimax(child, depth-1, True))
        return value
